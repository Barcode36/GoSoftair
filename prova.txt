con questo codice jquery
			
			$(document).ready(function(){
			$('a#delete').click(function(){
				$('#finestra').dialog("open");
				return false;	
			});
			
			$('#finestra').dialog({
				modal: true,
				autoOpen: false,
				width: 450,
				buttons: {
				"Annulla": function() {
					$( this ).dialog( "close" );
				},
				'Procedi': function(){
					$( this ).dialog( "close" );
					$("#form").submit();
					}
				}

			});	
		});
